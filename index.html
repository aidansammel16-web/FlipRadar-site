  <script>

    const FUNCTION_URL = "https://dyznxulsbhkhougueusp.supabase.co/functions/v1/create-checkout-session";

    const SITE_URL = "https://aidansammel16-web.github.io/FlipRadar-site/";


    function getUserIdFromUrl() {

      const params = new URLSearchParams(window.location.search);

      return params.get("user_id");

    }


    function showCheckoutStatus() {

      const params = new URLSearchParams(window.location.search);

      const main = document.querySelector("main");


      if (!main) return;


      if (params.get("success") === "true") {

        const banner = document.createElement("div");

        banner.textContent = "✅ You are now Pro. Reopen the extension and your Pro features should unlock within a few seconds.";

        banner.style.maxWidth = "1100px";

        banner.style.margin = "24px auto 0";

        banner.style.padding = "14px 18px";

        banner.style.borderRadius = "10px";

        banner.style.background = "#dcfce7";

        banner.style.color = "#166534";

        banner.style.fontWeight = "600";

        banner.style.border = "1px solid #86efac";

        main.prepend(banner);

        window.scrollTo({ top: 0, behavior: "smooth" });

      }


      if (params.get("canceled") === "true") {

        const banner = document.createElement("div");

        banner.textContent = "Payment was canceled. You can try again anytime.";

        banner.style.maxWidth = "1100px";

        banner.style.margin = "24px auto 0";

        banner.style.padding = "14px 18px";

        banner.style.borderRadius = "10px";

        banner.style.background = "#fef3c7";

        banner.style.color = "#92400e";

        banner.style.fontWeight = "600";

        banner.style.border = "1px solid #fcd34d";

        main.prepend(banner);

        window.scrollTo({ top: 0, behavior: "smooth" });

      }

    }


    async function startCheckout() {

      const user_id = getUserIdFromUrl();


      if (!user_id) {

        alert("Missing user ID. Please open this page from the FlipRadar extension.");

        return;

      }


      try {

        const res = await fetch(FUNCTION_URL, {

          method: "POST",

          headers: {

            "Content-Type": "application/json"

          },

          body: JSON.stringify({

            user_id,

            success_url: `${SITE_URL}?user_id=${encodeURIComponent(user_id)}&success=true`,

            cancel_url: `${SITE_URL}?user_id=${encodeURIComponent(user_id)}&canceled=true`

          })

        });


        const raw = await res.text();

        let data = {};


        try {

          data = JSON.parse(raw);

        } catch (e) {

          console.error("Non-JSON checkout response:", raw);

        }


        if (!res.ok) {

          console.error("Checkout function failed:", res.status, data || raw);

          alert("Checkout failed. Check the console for details.");

          return;

        }


        if (data.url) {

          window.location.href = data.url;

        } else {

          console.error("Checkout response missing URL:", data);

          alert("Checkout failed");

        }

      } catch (err) {

        alert("Checkout failed");

        console.error("Checkout request error:", err);

      }

    }


    showCheckoutStatus();


    document.getElementById("goProBtnHero").addEventListener("click", startCheckout);

    document.getElementById("goProBtnPricing").addEventListener("click", startCheckout);

  </script>
